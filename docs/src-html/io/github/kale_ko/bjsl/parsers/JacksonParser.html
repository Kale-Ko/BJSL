024/**
025 * An abstract class that all Jackson parsers extend from
026 * <p>
027 * Contains all the logic for parsing, factories are just passed for values
028 *
029 * @param <T> The type of the factory used for converting to/from trees/strings
030 * @param <V> The type of the codec used for converting to/from trees/strings
031 *
032 * @version 2.0.0
033 * @since 1.0.0
034 */
035public abstract class JacksonParser<T extends TokenStreamFactory, V extends ObjectCodec> extends Parser {
036    /**
037     * The factory used for converting to/from trees/strings
038     *
039     * @since 1.0.0
040     */
041    protected final @NotNull T factory;
042
043    /**
044     * The codec used for converting to/from trees/strings
045     *
046     * @since 1.0.0
047     */
048    protected final @NotNull V codec;
049
050    /**
051     * The prettyPrinter used for converting to strings
052     *
053     * @since 1.0.0
054     */
055    protected final @Nullable PrettyPrinter prettyPrinter;
056
057    /**
058     * Create a new JacksonParser using certain factories
059     *
060     * @param factory       The factory used for converting to/from trees/strings
061     * @param codec         The codec used for converting to/from trees/strings
062     * @param prettyPrinter The prettyPrinter used for converting to strings
063     *
064     * @since 1.0.0
065     */
066    protected JacksonParser(@NotNull T factory, @NotNull V codec, @Nullable PrettyPrinter prettyPrinter) {
067        this.factory = factory;
068        this.codec = codec;
069
070        this.prettyPrinter = prettyPrinter;
071    }
072
073    /**
074     * {@inheritDoc}
075     *
076     * @implNote Calls {@link #toElement(byte[])} with the bytes of the passed string ({@link String#getBytes()})
077     */
078    @Override
079    public @NotNull ParsedElement toElement(@NotNull String data) {
080        return toElement(data.getBytes(StandardCharsets.UTF_8));
081    }
082
083    /**
084     * {@inheritDoc}
085     */
086    @Override
087    public @NotNull ParsedElement toElement(byte @NotNull [] data) {
088        try {
089            TreeNode tree;
090            try (com.fasterxml.jackson.core.JsonParser parser = this.factory.createParser(data)) {
091                parser.setCodec(this.codec);
092                tree = parser.readValueAsTree();
093            }
094
095            if (tree instanceof JsonNode) {
096                JsonNode node = (JsonNode) tree;
097
098                switch (node.getNodeType()) {
099                    case OBJECT: {
100                        ObjectNode objectNode = (ObjectNode) node;
101                        ParsedObject objectElement = ParsedObject.create();
102
103                        objectNode.fields().forEachRemaining(subEntry -> {
104                            toElements(objectElement, subEntry.getKey(), subEntry.getValue());
105                        });
106
107                        return objectElement;
108                    }
109                    case ARRAY: {
110                        ArrayNode arrayNode = (ArrayNode) node;
111                        ParsedArray arrayElement = ParsedArray.create();
112
113                        arrayNode.elements().forEachRemaining(subNode -> {
114                            toElements(arrayElement, "root", subNode);
115                        });
116
117                        return arrayElement;
118                    }
119                    case STRING: {
120                        return ParsedPrimitive.fromString(node.asText());
121                    }
122                    case NUMBER: {
123                        switch (node.numberType()) {
124                            case INT: {
125                                return ParsedPrimitive.fromInteger(node.asInt());
126                            }
127                            case LONG: {
128                                return ParsedPrimitive.fromLong(node.asLong());
129                            }
130                            case BIG_INTEGER: {
131                                return ParsedPrimitive.fromBigInteger(node.bigIntegerValue());
132                            }
133                            case FLOAT: {
134                                return ParsedPrimitive.fromFloat((float) node.asDouble());
135                            }
136                            case DOUBLE: {
137                                return ParsedPrimitive.fromDouble(node.asDouble());
138                            }
139                            case BIG_DECIMAL: {
140                                return ParsedPrimitive.fromBigDecimal(node.decimalValue());
141                            }
142                            default: {
143                                throw new InvalidTypeException(node.getClass());
144                            }
145                        }
146                    }
147                    case BOOLEAN: {
148                        return ParsedPrimitive.fromBoolean(node.asBoolean());
149                    }
150                    case NULL: {
151                        return ParsedPrimitive.fromNull();
152                    }
153                    default: {
154                        throw new InvalidTypeException(node.getClass());
155                    }
156                }
157            } else {
158                throw new InvalidTypeException(tree.getClass());
159            }
160        } catch (Exception e) {
161            throw new ParserException(e);
162        }
163    }
164
165    /**
166     * {@inheritDoc}
167     *
168     * @implNote Calls {@link #toBytes(ParsedElement)} and uses {@link String#String(byte[])} to create a string from that
169     */
170    @Override
171    public @NotNull String toString(@NotNull ParsedElement element) {
172        return new String(toBytes(element), StandardCharsets.UTF_8);
173    }
174
175    /**
176     * {@inheritDoc}
177     */
178    @Override
179    public byte @NotNull [] toBytes(@NotNull ParsedElement element) {
180        try {
181            if (element instanceof ParsedObject) {
182                ParsedObject objectElement = (ParsedObject) element;
183
184                ObjectNode node = JsonNodeFactory.instance.objectNode();
185                for (Map.Entry<String, ParsedElement> subElement : objectElement.getEntries()) {
186                    toNodes(node, subElement.getKey(), subElement.getValue());
187                }
188
189                ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
190                try (com.fasterxml.jackson.core.JsonGenerator generator = this.factory.createGenerator(outputStream).setPrettyPrinter(this.prettyPrinter)) {
191                    generator.setCodec(this.codec);
192                    generator.writeTree(node);
193                }
194                outputStream.close();
195
196                return outputStream.toByteArray();
197            } else if (element instanceof ParsedArray) {
198                ParsedArray arrayElement = (ParsedArray) element;
199
200                ArrayNode node = JsonNodeFactory.instance.arrayNode();
201                for (ParsedElement subElement : arrayElement.getValues()) {
202                    toNodes(node, "root", subElement);
203                }
204
205                ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
206                try (com.fasterxml.jackson.core.JsonGenerator generator = this.factory.createGenerator(outputStream).setPrettyPrinter(this.prettyPrinter)) {
207                    generator.setCodec(this.codec);
208                    generator.writeTree(node);
209                }
210                outputStream.close();
211
212                return outputStream.toByteArray();
213            } else if (element instanceof ParsedPrimitive) {
214                Object value = element.asPrimitive().get();
215                return (value != null ? value.toString() : "null").getBytes(StandardCharsets.UTF_8);
216            } else {
217                throw new InvalidTypeException(element.getClass());
218            }
219        } catch (Exception e) {
220            throw new ParserException(e);
221        }
222    }

249     * Serializes an empty object element into bytes
250     *
251     * @return The bytes for a new/empty object
252     *
253     * @since 1.3.0
254     */
255    public byte @NotNull [] emptyBytes() {
256        try {
257            ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
258            try (com.fasterxml.jackson.core.JsonGenerator generator = this.factory.createGenerator(outputStream).setPrettyPrinter(this.prettyPrinter)) {
259                generator.setCodec(this.codec);
260                generator.writeTree(JsonNodeFactory.instance.objectNode());
261            }
262            outputStream.close();
263
264            return outputStream.toByteArray();
265        } catch (Exception e) {
266            throw new ParserException(e);
267        }

292    /**
293     * Converts Jackson's format to BJSL's format
294     *
295     * @param element The element to add too
296     * @param key     The key of the element
297     * @param node    The node to convert
298     *
299     * @since 1.0.0
300     */
301    protected void toElements(@NotNull ParsedElement element, @NotNull String key, @NotNull JsonNode node) {
302        switch (node.getNodeType()) {
303            case OBJECT: {
304                ObjectNode objectNode = (ObjectNode) node;
305                ParsedObject subElement = ParsedObject.create();
306
307                if (element instanceof ParsedObject) {
308                    ((ParsedObject) element).set(key, subElement);
309                } else if (element instanceof ParsedArray) {
310                    ((ParsedArray) element).add(subElement);
311                }
312
313                objectNode.fields().forEachRemaining(subEntry -> {
314                    toElements(subElement, subEntry.getKey(), subEntry.getValue());
315                });
316
317                break;
318            }
319            case ARRAY: {
320                ArrayNode arrayNode = (ArrayNode) node;
321                ParsedArray subElement = ParsedArray.create();
322
323                if (element instanceof ParsedObject) {
324                    ((ParsedObject) element).set(key, subElement);
325                } else if (element instanceof ParsedArray) {
326                    ((ParsedArray) element).add(subElement);
327                }
328
329                arrayNode.elements().forEachRemaining(subNode -> {
330                    toElements(subElement, key, subNode);
331                });
332
333                break;
334            }
335            case STRING: {
336                if (element instanceof ParsedObject) {
337                    ((ParsedObject) element).set(key, ParsedPrimitive.fromString(node.asText()));
338                } else if (element instanceof ParsedArray) {
339                    ((ParsedArray) element).add(ParsedPrimitive.fromString(node.asText()));
340                }
341                break;
342            }
343            case NUMBER: {
344                switch (node.numberType()) {
345                    case INT: {
346                        if (element instanceof ParsedObject) {
347                            ((ParsedObject) element).set(key, ParsedPrimitive.fromInteger(node.asInt()));
348                        } else if (element instanceof ParsedArray) {
349                            ((ParsedArray) element).add(ParsedPrimitive.fromInteger(node.asInt()));
350                        }
351                        break;
352                    }
353                    case LONG: {
354                        if (element instanceof ParsedObject) {
355                            ((ParsedObject) element).set(key, ParsedPrimitive.fromLong(node.asLong()));
356                        } else if (element instanceof ParsedArray) {
357                            ((ParsedArray) element).add(ParsedPrimitive.fromLong(node.asLong()));
358                        }
359                        break;
360                    }
361                    case BIG_INTEGER: {
362                        if (element instanceof ParsedObject) {
363                            ((ParsedObject) element).set(key, ParsedPrimitive.fromBigInteger(node.bigIntegerValue()));
364                        } else if (element instanceof ParsedArray) {
365                            ((ParsedArray) element).add(ParsedPrimitive.fromBigInteger(node.bigIntegerValue()));
366                        }
367                        break;
368                    }
369                    case FLOAT: {
370                        if (element instanceof ParsedObject) {
371                            ((ParsedObject) element).set(key, ParsedPrimitive.fromFloat((float) node.asDouble()));
372                        } else if (element instanceof ParsedArray) {
373                            ((ParsedArray) element).add(ParsedPrimitive.fromFloat((float) node.asDouble()));
374                        }
375                        break;
376                    }
377                    case DOUBLE: {
378                        if (element instanceof ParsedObject) {
379                            ((ParsedObject) element).set(key, ParsedPrimitive.fromDouble(node.asDouble()));
380                        } else if (element instanceof ParsedArray) {
381                            ((ParsedArray) element).add(ParsedPrimitive.fromDouble(node.asDouble()));
382                        }
383                        break;
384                    }
385                    case BIG_DECIMAL: {
386                        if (element instanceof ParsedObject) {
387                            ((ParsedObject) element).set(key, ParsedPrimitive.fromBigDecimal(node.decimalValue()));
388                        } else if (element instanceof ParsedArray) {
389                            ((ParsedArray) element).add(ParsedPrimitive.fromBigDecimal(node.decimalValue()));
390                        }
391                        break;
392                    }
393                }
394                break;
395            }
396            case BOOLEAN: {
397                if (element instanceof ParsedObject) {
398                    ((ParsedObject) element).set(key, ParsedPrimitive.fromBoolean(node.asBoolean()));
399                } else if (element instanceof ParsedArray) {
400                    ((ParsedArray) element).add(ParsedPrimitive.fromBoolean(node.asBoolean()));
401                }
402                break;
403            }
404            case NULL: {
405                if (element instanceof ParsedObject) {
406                    ((ParsedObject) element).set(key, ParsedPrimitive.fromNull());
407                } else if (element instanceof ParsedArray) {
408                    ((ParsedArray) element).add(ParsedPrimitive.fromNull());
409                }
410                break;
411            }
412            default: {
413                if (BJSL.getLogger() != null) {
414                    BJSL.getLogger().warning("Warning while parsing: Node \"" + key + "\" of type \"" + node.getClass().getSimpleName() + "\" is not serializable");
415                }
416                break;
417            }
418        }
419    }

421    /**
422     * Converts BJSL's format to Jackson's format
423     *
424     * @param element The element to add too
425     * @param key     The key of the element
426     * @param node    The node to convert
427     *
428     * @since 1.0.0
429     */
430    protected void toNodes(@NotNull TreeNode node, @NotNull String key, @NotNull ParsedElement element) {
431        if (element instanceof ParsedObject) {
432            ParsedObject objectElement = (ParsedObject) element;
433            ObjectNode subNode = JsonNodeFactory.instance.objectNode();
434
435            if (node instanceof ObjectNode) {
436                ((ObjectNode) node).set(key, subNode);
437            } else if (node instanceof ArrayNode) {
438                ((ArrayNode) node).add(subNode);
439            }
440
441            for (Map.Entry<String, ParsedElement> subElement : objectElement.getEntries()) {
442                toNodes(subNode, subElement.getKey(), subElement.getValue());
443            }
444        } else if (element instanceof ParsedArray) {
445            ParsedArray arrayElement = (ParsedArray) element;
446            ArrayNode subNode = JsonNodeFactory.instance.arrayNode();
447
448            if (node instanceof ObjectNode) {
449                ((ObjectNode) node).set(key, subNode);
450            } else if (node instanceof ArrayNode) {
451                ((ArrayNode) node).add(subNode);
452            }
453
454            for (ParsedElement subElement : arrayElement.getValues()) {
455                toNodes(subNode, key, subElement);
456            }
457        } else if (element instanceof ParsedPrimitive) {
458            ParsedPrimitive primitiveElement = (ParsedPrimitive) element;
459
460            switch (primitiveElement.getType()) {
461                case STRING: {
462                    if (node instanceof ObjectNode) {
463                        ((ObjectNode) node).put(key, primitiveElement.asString());
464                    } else if (node instanceof ArrayNode) {
465                        ((ArrayNode) node).add(primitiveElement.asString());
466                    }
467                    break;
468                }
469                case BYTE: {
470                    if (node instanceof ObjectNode) {
471                        ((ObjectNode) node).put(key, primitiveElement.asByte());
472                    } else if (node instanceof ArrayNode) {
473                        ((ArrayNode) node).add(primitiveElement.asByte());
474                    }
475                    break;
476                }
477                case CHAR: {
478                    if (node instanceof ObjectNode) {
479                        ((ObjectNode) node).put(key, primitiveElement.asChar());
480                    } else if (node instanceof ArrayNode) {
481                        ((ArrayNode) node).add(primitiveElement.asChar());
482                    }
483                    break;
484                }
485                case SHORT: {
486                    if (node instanceof ObjectNode) {
487                        ((ObjectNode) node).put(key, primitiveElement.asShort());
488                    } else if (node instanceof ArrayNode) {
489                        ((ArrayNode) node).add(primitiveElement.asShort());
490                    }
491                    break;
492                }
493                case INTEGER: {
494                    if (node instanceof ObjectNode) {
495                        ((ObjectNode) node).put(key, primitiveElement.asInteger());
496                    } else if (node instanceof ArrayNode) {
497                        ((ArrayNode) node).add(primitiveElement.asInteger());
498                    }
499                    break;
500                }
501                case LONG: {
502                    if (node instanceof ObjectNode) {
503                        ((ObjectNode) node).put(key, primitiveElement.asLong());
504                    } else if (node instanceof ArrayNode) {
505                        ((ArrayNode) node).add(primitiveElement.asLong());
506                    }
507                    break;
508                }
509                case BIGINTEGER: {
510                    if (node instanceof ObjectNode) {
511                        ((ObjectNode) node).put(key, primitiveElement.asBigInteger());
512                    } else if (node instanceof ArrayNode) {
513                        ((ArrayNode) node).add(primitiveElement.asBigInteger());
514                    }
515                    break;
516                }
517                case FLOAT: {
518                    if (node instanceof ObjectNode) {
519                        ((ObjectNode) node).put(key, primitiveElement.asFloat());
520                    } else if (node instanceof ArrayNode) {
521                        ((ArrayNode) node).add(primitiveElement.asFloat());
522                    }
523                    break;
524                }
525                case DOUBLE: {
526                    if (node instanceof ObjectNode) {
527                        ((ObjectNode) node).put(key, primitiveElement.asDouble());
528                    } else if (node instanceof ArrayNode) {
529                        ((ArrayNode) node).add(primitiveElement.asDouble());
530                    }
531                    break;
532                }
533                case BIGDECIMAL: {
534                    if (node instanceof ObjectNode) {
535                        ((ObjectNode) node).put(key, primitiveElement.asBigDecimal());
536                    } else if (node instanceof ArrayNode) {
537                        ((ArrayNode) node).add(primitiveElement.asBigDecimal());
538                    }
539                    break;
540                }
541                case BOOLEAN: {
542                    if (node instanceof ObjectNode) {
543                        ((ObjectNode) node).put(key, primitiveElement.asBoolean());
544                    } else if (node instanceof ArrayNode) {
545                        ((ArrayNode) node).add(primitiveElement.asBoolean());
546                    }
547                    break;
548                }
549                case NULL: {
550                    if (node instanceof ObjectNode) {
551                        ((ObjectNode) node).putNull(key);
552                    } else if (node instanceof ArrayNode) {
553                        ((ArrayNode) node).addNull();
554                    }
555                    break;
556                }
557                default: {
558                    if (BJSL.getLogger() != null) {
559                        BJSL.getLogger().warning("Warning while parsing: Element \"" + key + "\" of type \"" + primitiveElement.getType() + "\" is not serializable");
560                    }
561                    break;
562                }
563            }
564        } else {
565            if (BJSL.getLogger() != null) {
566                BJSL.getLogger().warning("Warning while parsing: Element \"" + key + "\" of type \"" + element.getClass().getSimpleName() + "\" is not serializable");
567            }
568        }
569    }
