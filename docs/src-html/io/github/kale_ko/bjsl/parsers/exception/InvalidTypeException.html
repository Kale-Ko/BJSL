001package io.github.kale_ko.bjsl.parsers.exception;
002
003import org.jetbrains.annotations.NotNull;
004
005/**
006 * Thrown when an invalid/unknown type is passed to a method that can't handle it
007 *
008 * @version 1.7.0
009 * @since 1.7.0
010 */
011public class InvalidTypeException extends RuntimeException {
012    /**
013     * Create a new InvalidTypeException
014     *
015     * @param type The type that was unable to be handled
016     */
017    public InvalidTypeException(@NotNull Class<?> type) {
018        super("\"" + type.getSimpleName() + "\" is not a processable type");
019    }
020}
