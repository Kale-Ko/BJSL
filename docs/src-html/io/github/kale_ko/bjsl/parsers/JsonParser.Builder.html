001package io.github.kale_ko.bjsl.parsers;
002
003import com.fasterxml.jackson.core.*;
004import com.fasterxml.jackson.core.json.JsonReadFeature;
005import com.fasterxml.jackson.core.json.JsonWriteFeature;
006import com.fasterxml.jackson.core.util.DefaultIndenter;
007import com.fasterxml.jackson.core.util.DefaultPrettyPrinter;
008import com.fasterxml.jackson.core.util.Separators;
009import com.fasterxml.jackson.databind.json.JsonMapper;
010import io.github.kale_ko.bjsl.BJSL;
011import java.io.PrintWriter;
012import java.io.StringWriter;
013import java.lang.reflect.Field;
014import org.jetbrains.annotations.NotNull;
015import org.jetbrains.annotations.Nullable;
016
017/**
018 * A parser for interfacing with JSON
019 * <p>
020 * Uses the default Jackson json parser
021 *
022 * @version 2.0.0
023 * @since 1.0.0
024 */
025public class JsonParser extends JacksonParser<JsonFactory, JsonMapper> {
026    /**
027     * Create a new JacksonParser using certain factories
028     *
029     * @param factory       The factory used for converting to/from trees/strings
030     * @param mapper        The mapper used for converting to/from trees/strings
031     * @param prettyPrinter The prettyPrinter used for converting to strings
032     *
033     * @since 1.0.0
034     */
035    protected JsonParser(@NotNull JsonFactory factory, @NotNull JsonMapper mapper, @Nullable PrettyPrinter prettyPrinter) {
036        super(factory, mapper, prettyPrinter);
037    }
038
039    /**
040     * A builder class for creating new {@link JsonParser}s
041     *
042     * @version 2.0.0
043     * @since 1.0.0
044     */
045    public static class Builder {
046        /**
047         * Weather pretty printing should be enabled or not
048         * <p>
049         * Default is false
050         *
051         * @since 1.0.0
052         */
053        protected boolean prettyPrint = false;
054
055        /**
056         * The indent level (in spaces) to use when pretty printing
057         * <p>
058         * Default is 2
059         *
060         * @since 1.0.0
061         */
062        protected int indentLevel = 2;
063
064        /**
065         * Weather to use crlf or lf line endings
066         * <p>
067         * Default is lf
068         *
069         * @since 1.0.0
070         */
071        protected boolean crlf = false;
072
073        /**
074         * Create a new {@link JsonParser} builder
075         *
076         * @since 1.0.0
077         */
078        public Builder() {
079        }
080
081        /**
082         * Get weather pretty printing should be enabled or not
083         * <p>
084         * Default is false
085         *
086         * @return Weather pretty printing should be enabled or not
087         *
088         * @since 1.0.0
089         */
090        public boolean getPrettyPrint() {
091            return this.prettyPrint;
092        }
093
094        /**
095         * Set weather pretty printing should be enabled or not
096         * <p>
097         * Default is false
098         *
099         * @param value Weather pretty printing should be enabled or not
100         *
101         * @return Self for chaining
102         *
103         * @since 1.0.0
104         */
105        public @NotNull Builder setPrettyPrint(boolean value) {
106            this.prettyPrint = value;
107
108            return this;
109        }
110
111        /**
112         * Get the indent level (in spaces) to use when pretty printing
113         * <p>
114         * Default is 2
115         *
116         * @return The indent level (in spaces) to use when pretty printing
117         *
118         * @since 1.0.0
119         */
120        public int getIndentLevel() {
121            return this.indentLevel;
122        }
123
124        /**
125         * Set the indent level (in spaces) to use when pretty printing
126         * <p>
127         * Default is 2
128         *
129         * @param value The indent level (in spaces) to use when pretty printing
130         *
131         * @return Self for chaining
132         *
133         * @since 1.0.0
134         */
135        public @NotNull Builder setIndentLevel(int value) {
136            this.indentLevel = value;
137
138            return this;
139        }
140
141        /**
142         * Get weather to use crlf or lf line endings
143         * <p>
144         * Default is lf
145         *
146         * @return Weather to use crlf or lf line endings
147         *
148         * @since 1.0.0
149         */
150        public boolean getCrlf() {
151            return this.crlf;
152        }
153
154        /**
155         * Set weather to use crlf or lf line endings
156         * <p>
157         * Default is lf
158         *
159         * @param value Weather to use crlf or lf line endings
160         *
161         * @return Self for chaining
162         *
163         * @since 1.0.0
164         */
165        public @NotNull Builder setCrlf(boolean value) {
166            this.crlf = value;
167
168            return this;
169        }
170
171        /**
172         * Uses the current settings to build a new {@link JsonParser}
173         *
174         * @return A new {@link JsonParser} instance
175         *
176         * @since 1.0.0
177         */
178        public @NotNull JsonParser build() {
179            JsonFactoryBuilder factoryBuilder = (JsonFactoryBuilder) JsonFactory.builder();
180            factoryBuilder = factoryBuilder.configure(StreamReadFeature.STRICT_DUPLICATE_DETECTION, true);
181            factoryBuilder = factoryBuilder.configure(StreamReadFeature.USE_FAST_DOUBLE_PARSER, true);
182            factoryBuilder = factoryBuilder.configure(StreamReadFeature.USE_FAST_BIG_NUMBER_PARSER, true);
183            factoryBuilder = factoryBuilder.configure(StreamWriteFeature.USE_FAST_DOUBLE_WRITER, true);
184            factoryBuilder = factoryBuilder.configure(StreamWriteFeature.WRITE_BIGDECIMAL_AS_PLAIN, false);
185            factoryBuilder = factoryBuilder.configure(JsonReadFeature.ALLOW_JAVA_COMMENTS, true);
186            factoryBuilder = factoryBuilder.configure(JsonReadFeature.ALLOW_YAML_COMMENTS, true);
187            factoryBuilder = factoryBuilder.configure(JsonReadFeature.ALLOW_MISSING_VALUES, true);
188            factoryBuilder = factoryBuilder.configure(JsonReadFeature.ALLOW_SINGLE_QUOTES, true);
189            factoryBuilder = factoryBuilder.configure(JsonReadFeature.ALLOW_BACKSLASH_ESCAPING_ANY_CHARACTER, true);
190            factoryBuilder = factoryBuilder.configure(JsonReadFeature.ALLOW_LEADING_DECIMAL_POINT_FOR_NUMBERS, true);
191            factoryBuilder = factoryBuilder.configure(JsonReadFeature.ALLOW_TRAILING_DECIMAL_POINT_FOR_NUMBERS, false);
192            factoryBuilder = factoryBuilder.configure(JsonReadFeature.ALLOW_LEADING_PLUS_SIGN_FOR_NUMBERS, true);
193            factoryBuilder = factoryBuilder.configure(JsonReadFeature.ALLOW_LEADING_ZEROS_FOR_NUMBERS, false);
194            factoryBuilder = factoryBuilder.configure(JsonReadFeature.ALLOW_NON_NUMERIC_NUMBERS, true);
195            factoryBuilder = factoryBuilder.configure(JsonReadFeature.ALLOW_TRAILING_COMMA, true);
196            factoryBuilder = factoryBuilder.configure(JsonWriteFeature.ESCAPE_NON_ASCII, true);
197            factoryBuilder = factoryBuilder.configure(JsonWriteFeature.WRITE_HEX_UPPER_CASE, true);
198            factoryBuilder = factoryBuilder.configure(JsonWriteFeature.WRITE_NAN_AS_STRINGS, false);
199            factoryBuilder = factoryBuilder.configure(JsonWriteFeature.COMBINE_UNICODE_SURROGATES_IN_UTF8, true);
200
201            JsonFactory factory = factoryBuilder.build();
202
203            DefaultPrettyPrinter prettyPrinter = new DefaultPrettyPrinter();
204
205            if (prettyPrint) {
206                DefaultIndenter indenter = new DefaultIndenter();
207                indenter = indenter.withIndent(" ".repeat(this.indentLevel));
208                indenter = indenter.withLinefeed(this.indentLevel > 0 ? (this.crlf ? "\r\n" : "\n") : "");
209
210                Separators separators = new Separators();
211                separators = separators.withObjectFieldValueSpacing(Separators.Spacing.AFTER);
212                separators = separators.withObjectEntrySpacing(this.indentLevel <= 0 ? Separators.Spacing.AFTER : Separators.Spacing.NONE);
213                separators = separators.withArrayValueSpacing(this.indentLevel <= 0 ? Separators.Spacing.AFTER : Separators.Spacing.NONE);
214                separators = separators.withObjectEmptySeparator(" ");
215                separators = separators.withArrayEmptySeparator(" ");
216
217                prettyPrinter = prettyPrinter.withObjectIndenter(indenter).withArrayIndenter(indenter).withSeparators(separators);
218
219                try {
220                    Field separatorField = prettyPrinter.getClass().getDeclaredField("_objectFieldValueSeparatorWithSpaces");
221                    separatorField.setAccessible(true);
222                    separatorField.set(prettyPrinter, ": ");
223                } catch (NoSuchFieldException | IllegalArgumentException | IllegalAccessException e) {
224                    if (BJSL.getLogger() != null) {
225                        StringWriter writer = new StringWriter();
226                        new RuntimeException("Error while parsing:", e).printStackTrace(new PrintWriter(writer));
227                        BJSL.getLogger().severe(writer.toString());
228                    }
229                }
230            } else {
231                DefaultIndenter indenter = new DefaultIndenter();
232                indenter = indenter.withIndent("");
233                indenter = indenter.withLinefeed("");
234
235                Separators separators = new Separators();
236                separators = separators.withObjectFieldValueSpacing(Separators.Spacing.NONE);
237                separators = separators.withObjectEntrySpacing(Separators.Spacing.NONE);
238                separators = separators.withArrayValueSpacing(Separators.Spacing.NONE);
239
240                prettyPrinter = prettyPrinter.withObjectIndenter(indenter).withArrayIndenter(indenter).withSeparators(separators);
241            }
242
243            return new JsonParser(factory, new JsonMapper(factory), prettyPrinter);
244        }
245    }
246}
