001package io.github.kale_ko.bjsl.parsers;
002
003import com.fasterxml.jackson.core.PrettyPrinter;
004import com.fasterxml.jackson.core.StreamReadFeature;
005import com.fasterxml.jackson.core.StreamWriteFeature;
006import com.fasterxml.jackson.dataformat.smile.SmileFactory;
007import com.fasterxml.jackson.dataformat.smile.SmileFactoryBuilder;
008import com.fasterxml.jackson.dataformat.smile.SmileGenerator;
009import com.fasterxml.jackson.dataformat.smile.databind.SmileMapper;
010import org.jetbrains.annotations.NotNull;
011import org.jetbrains.annotations.Nullable;
012
013/**
014 * A parser for interfacing with Smile (Binary JSON)
015 * <p>
016 * Uses the Jackson-DataFormat smile parser
017 *
018 * @version 2.0.0
019 * @since 1.0.0
020 */
021public class SmileParser extends JacksonParser<SmileFactory, SmileMapper> {
022    /**
023     * Create a new JacksonParser using certain factories
024     *
025     * @param factory       The factory used for converting to/from trees/strings
026     * @param mapper        The mapper used for converting to/from trees/strings
027     * @param prettyPrinter The prettyPrinter used for converting to strings
028     *
029     * @since 1.0.0
030     */
031    protected SmileParser(@NotNull SmileFactory factory, @NotNull SmileMapper mapper, @Nullable PrettyPrinter prettyPrinter) {
032        super(factory, mapper, prettyPrinter);
033    }
034
035    /**
036     * A builder class for creating new {@link SmileParser}s
037     *
038     * @version 2.0.0
039     * @since 1.0.0
040     */
041    public static class Builder {
042        /**
043         * Create a new {@link SmileParser} builder
044         *
045         * @since 1.0.0
046         */
047        public Builder() {
048        }
049
050        /**
051         * Uses the current settings to build a new {@link SmileParser}
052         *
053         * @return A new {@link SmileParser} instance
054         *
055         * @since 1.0.0
056         */
057        public @NotNull SmileParser build() {
058            SmileFactoryBuilder factoryBuilder = SmileFactory.builder();
059            factoryBuilder = factoryBuilder.configure(StreamReadFeature.STRICT_DUPLICATE_DETECTION, true);
060            factoryBuilder = factoryBuilder.configure(StreamReadFeature.USE_FAST_DOUBLE_PARSER, true);
061            factoryBuilder = factoryBuilder.configure(StreamReadFeature.USE_FAST_BIG_NUMBER_PARSER, true);
062            factoryBuilder = factoryBuilder.configure(StreamWriteFeature.USE_FAST_DOUBLE_WRITER, true);
063            factoryBuilder = factoryBuilder.configure(StreamWriteFeature.WRITE_BIGDECIMAL_AS_PLAIN, false);
064            factoryBuilder = factoryBuilder.configure(SmileGenerator.Feature.WRITE_HEADER, true);
065            factoryBuilder = factoryBuilder.configure(SmileGenerator.Feature.WRITE_END_MARKER, true);
066            factoryBuilder = factoryBuilder.configure(SmileGenerator.Feature.CHECK_SHARED_NAMES, true);
067            factoryBuilder = factoryBuilder.configure(SmileGenerator.Feature.CHECK_SHARED_STRING_VALUES, false);
068
069            SmileFactory factory = factoryBuilder.build();
070
071            return new SmileParser(factory, new SmileMapper(factory), null);
072        }
073    }
074}
