001package io.github.kale_ko.bjsl.parsers;
002
003import com.fasterxml.jackson.core.PrettyPrinter;
004import com.fasterxml.jackson.core.StreamReadFeature;
005import com.fasterxml.jackson.core.StreamWriteFeature;
006import com.fasterxml.jackson.core.util.DefaultIndenter;
007import com.fasterxml.jackson.core.util.DefaultPrettyPrinter;
008import com.fasterxml.jackson.core.util.Separators;
009import com.fasterxml.jackson.dataformat.yaml.*;
010import org.jetbrains.annotations.NotNull;
011import org.jetbrains.annotations.Nullable;
012import org.yaml.snakeyaml.DumperOptions;
013
014/**
015 * A parser for interfacing with YAML
016 * <p>
017 * Uses the Jackson-DataFormat yaml parser
018 *
019 * @version 2.0.0
020 * @since 1.0.0
021 */
022public class YamlParser extends JacksonParser<YAMLFactory, YAMLMapper> {
023    /**
024     * Create a new JacksonParser using certain factories
025     *
026     * @param factory       The factory used for converting to/from trees/strings
027     * @param mapper        The mapper used for converting to/from trees/strings
028     * @param prettyPrinter The prettyPrinter used for converting to strings
029     *
030     * @since 1.0.0
031     */
032    protected YamlParser(@NotNull YAMLFactory factory, @NotNull YAMLMapper mapper, @Nullable PrettyPrinter prettyPrinter) {
033        super(factory, mapper, prettyPrinter);
034    }
035
036    /**
037     * A builder class for creating new {@link YamlParser}s
038     *
039     * @version 2.0.0
040     * @since 1.0.0
041     */
042    public static class Builder {
043        /**
044         * The indent level (in spaces) to use when pretty printing
045         * <p>
046         * Default is 2
047         *
048         * @since 1.0.0
049         */
050        protected int indentLevel = 2;
051
052        /**
053         * Weather to use crlf or lf line endings
054         * <p>
055         * Default is lf
056         *
057         * @since 1.0.0
058         */
059        protected boolean crlf = false;
060
061        /**
062         * Create a new {@link YamlParser} builder
063         *
064         * @since 1.0.0
065         */
066        public Builder() {
067        }
068
069        /**
070         * Get the indent level (in spaces) to use when pretty printing
071         * <p>
072         * Default is 2
073         *
074         * @return The indent level (in spaces) to use when pretty printing
075         *
076         * @since 1.0.0
077         */
078        public int getIndentLevel() {
079            return this.indentLevel;
080        }
081
082        /**
083         * Set the indent level (in spaces) to use when pretty printing
084         * <p>
085         * Default is 2
086         *
087         * @param value The indent level (in spaces) to use when pretty printing
088         *
089         * @return Self for chaining
090         *
091         * @since 1.0.0
092         */
093        public @NotNull Builder setIndentLevel(int value) {
094            this.indentLevel = value;
095
096            return this;
097        }
098
099        /**
100         * Get weather to use crlf or lf line endings
101         * <p>
102         * Default is lf
103         *
104         * @return Weather to use crlf or lf line endings
105         *
106         * @since 1.0.0
107         */
108        public boolean getCrlf() {
109            return this.crlf;
110        }
111
112        /**
113         * Set weather to use crlf or lf line endings
114         * <p>
115         * Default is lf
116         *
117         * @param value Weather to use crlf or lf line endings
118         *
119         * @return Self for chaining
120         *
121         * @since 1.0.0
122         */
123        public @NotNull Builder setCrlf(boolean value) {
124            this.crlf = value;
125
126            return this;
127        }
128
129        /**
130         * Uses the current settings to build a new {@link YamlParser}
131         *
132         * @return A new {@link YamlParser} instance
133         *
134         * @since 1.0.0
135         */
136        public @NotNull YamlParser build() {
137            YAMLFactoryBuilder factoryBuilder = YAMLFactory.builder();
138            factoryBuilder = factoryBuilder.configure(StreamReadFeature.STRICT_DUPLICATE_DETECTION, true);
139            factoryBuilder = factoryBuilder.configure(StreamReadFeature.USE_FAST_DOUBLE_PARSER, true);
140            factoryBuilder = factoryBuilder.configure(StreamReadFeature.USE_FAST_BIG_NUMBER_PARSER, true);
141            factoryBuilder = factoryBuilder.configure(StreamWriteFeature.USE_FAST_DOUBLE_WRITER, true);
142            factoryBuilder = factoryBuilder.configure(StreamWriteFeature.WRITE_BIGDECIMAL_AS_PLAIN, false);
143            factoryBuilder = factoryBuilder.configure(YAMLGenerator.Feature.WRITE_DOC_START_MARKER, false);
144            factoryBuilder = factoryBuilder.configure(YAMLGenerator.Feature.ALLOW_LONG_KEYS, true);
145            factoryBuilder = factoryBuilder.configure(YAMLGenerator.Feature.SPLIT_LINES, false);
146            factoryBuilder = factoryBuilder.configure(YAMLGenerator.Feature.MINIMIZE_QUOTES, false);
147            factoryBuilder = factoryBuilder.configure(YAMLGenerator.Feature.INDENT_ARRAYS, true);
148            factoryBuilder = factoryBuilder.configure(YAMLGenerator.Feature.INDENT_ARRAYS_WITH_INDICATOR, true);
149            factoryBuilder = factoryBuilder.configure(YAMLGenerator.Feature.USE_PLATFORM_LINE_BREAKS, this.crlf);
150            factoryBuilder = factoryBuilder.configure(YAMLParser.Feature.EMPTY_STRING_AS_NULL, false);
151
152            DumperOptions dumperOptions = new DumperOptions();
153
154            dumperOptions.setIndent(this.indentLevel);
155            dumperOptions.setIndicatorIndent(this.indentLevel);
156            dumperOptions.setIndentWithIndicator(true);
157            dumperOptions.setLineBreak(this.crlf ? DumperOptions.LineBreak.WIN : DumperOptions.LineBreak.UNIX);
158            dumperOptions.setSplitLines(false);
159            dumperOptions.setDefaultScalarStyle(DumperOptions.ScalarStyle.LITERAL);
160
161            factoryBuilder.dumperOptions(dumperOptions);
162
163            YAMLFactory factory = factoryBuilder.build();
164
165            DefaultPrettyPrinter prettyPrinter = new DefaultPrettyPrinter();
166
167            DefaultIndenter indenter = new DefaultIndenter();
168            indenter = indenter.withIndent(" ".repeat(this.indentLevel));
169            indenter = indenter.withLinefeed(this.crlf ? "\r\n" : "\n");
170
171            Separators separators = new Separators();
172            separators = separators.withObjectFieldValueSpacing(Separators.Spacing.AFTER);
173            separators = separators.withObjectEntrySpacing(Separators.Spacing.AFTER);
174            separators = separators.withArrayValueSpacing(Separators.Spacing.AFTER);
175
176            prettyPrinter = prettyPrinter.withObjectIndenter(indenter).withArrayIndenter(indenter).withSeparators(separators);
177
178            return new YamlParser(factory, new YAMLMapper(factory), prettyPrinter);
179        }
180    }
181}
