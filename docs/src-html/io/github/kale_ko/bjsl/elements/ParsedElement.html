001package io.github.kale_ko.bjsl.elements;
002
003import org.jetbrains.annotations.NotNull;
004import org.jetbrains.annotations.Nullable;
005
006/**
007 * An abstract class that all element types extend from
008 *
009 * @version 1.0.0
010 * @since 1.0.0
011 */
012public abstract class ParsedElement {
013    /**
014     * Create a new {@link ParsedElement}
015     *
016     * @since 1.0.0
017     */
018    ParsedElement() {
019    }
020
021    @Override
022    public abstract @NotNull String toString();
023
024    @Override
025    public abstract boolean equals(@Nullable Object obj);
026
027    @Override
028    public abstract int hashCode();
029
030    /**
031     * Check if this element is a {@link ParsedObject}
032     *
033     * @return If the element is a {@link ParsedObject}
034     *
035     * @since 1.0.0
036     */
037    public boolean isObject() {
038        return this instanceof ParsedObject;
039    }
040
041    /**
042     * Check if this element is a {@link ParsedArray}
043     *
044     * @return If the element is a {@link ParsedArray}
045     *
046     * @since 1.0.0
047     */
048    public boolean isArray() {
049        return this instanceof ParsedArray;
050    }
051
052    /**
053     * Check if this element is a {@link ParsedPrimitive}
054     *
055     * @return If the element is a {@link ParsedPrimitive}
056     *
057     * @since 1.0.0
058     */
059    public boolean isPrimitive() {
060        return this instanceof ParsedPrimitive;
061    }
062
063    /**
064     * Cast this element to a {@link ParsedObject}
065     * <p>
066     * Note: Does not catch casting errors
067     *
068     * @return This element as a {@link ParsedObject}
069     *
070     * @throws java.lang.ClassCastException If this element is not an object
071     * @since 1.0.0
072     */
073    public @NotNull ParsedObject asObject() {
074        return (ParsedObject) this;
075    }
076
077    /**
078     * Cast this element to a {@link ParsedArray}
079     * <p>
080     * Note: Does not catch casting errors
081     *
082     * @return This element as a {@link ParsedArray}
083     *
084     * @throws java.lang.ClassCastException If this element is not an array
085     * @since 1.0.0
086     */
087    public @NotNull ParsedArray asArray() {
088        return (ParsedArray) this;
089    }
090
091    /**
092     * Cast this element to a {@link ParsedPrimitive}
093     * <p>
094     * Note: Does not catch casting errors
095     *
096     * @return This element as a {@link ParsedPrimitive}
097     *
098     * @throws java.lang.ClassCastException If this element is not a primitive
099     * @since 1.0.0
100     */
101    public @NotNull ParsedPrimitive asPrimitive() {
102        return (ParsedPrimitive) this;
103    }
104}
