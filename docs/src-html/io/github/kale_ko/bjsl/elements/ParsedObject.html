001package io.github.kale_ko.bjsl.elements;
002
003import java.util.Collection;
004import java.util.Collections;
005import java.util.LinkedHashMap;
006import java.util.Map;
007import org.jetbrains.annotations.NotNull;
008import org.jetbrains.annotations.Nullable;
009import org.jetbrains.annotations.Unmodifiable;
010
011/**
012 * A wrapper for an ordered map used to represent an Object in most data formats
013 *
014 * @version 2.0.0
015 * @since 1.0.0
016 */
017public class ParsedObject extends ParsedElement {
018    /**
019     * The map this object stores
020     *
021     * @since 1.0.0
022     */
023    final @NotNull Map<String, ParsedElement> object;
024
025    /**
026     * Create a new {@link ParsedObject} from an object
027     *
028     * @param object The object to use
029     *
030     * @since 1.0.0
031     */
032    ParsedObject(@NotNull Map<String, ParsedElement> object) {
033        this.object = object;
034    }
035
036    /**
037     * Get the amount of key/value pairs this object stores
038     *
039     * @return The amount of key/value pairs this object stores
040     *
041     * @since 1.0.0
042     */
043    public int getSize() {
044        return this.object.size();
045    }
046
047    /**
048     * Get a list of all the key/value pairs this object stores
049     * <p>
050     * Note: Returns a copy of the list
051     *
052     * @return A list of all the key/value pairs this object stores
053     *
054     * @since 1.0.0
055     */
056    public @NotNull @Unmodifiable Collection<Map.Entry<String, ParsedElement>> getEntries() {
057        return Collections.unmodifiableCollection(this.object.entrySet());
058    }
059
060    /**
061     * Get a list of all the keys this object stores
062     * <p>
063     * Note: Returns a copy of the list
064     *
065     * @return A list of all the keys this object stores
066     *
067     * @since 1.0.0
068     */
069    public @NotNull @Unmodifiable Collection<String> getKeys() {
070        return Collections.unmodifiableCollection(this.object.keySet());
071    }
072
073    /**
074     * Get a list of all the values this object stores
075     * <p>
076     * Note: Returns a copy of the list
077     *
078     * @return A list of all the values this object stores
079     *
080     * @since 1.0.0
081     */
082    public @NotNull @Unmodifiable Collection<ParsedElement> getValues() {
083        return Collections.unmodifiableCollection(this.object.values());
084    }
085
086    /**
087     * Check if this object stores a certain key
088     *
089     * @param key The key to check
090     *
091     * @return If this object stores a certain key
092     *
093     * @since 1.0.0
094     */
095    public boolean has(@NotNull String key) {
096        return this.object.containsKey(key);
097    }
098
099    /**
100     * Get the value of a certain key this object stores
101     *
102     * @param key The key to get
103     *
104     * @return The value of a certain key this object stores
105     *
106     * @throws java.lang.NullPointerException If there is no value associated with the key
107     * @since 1.0.0
108     */
109    public @NotNull ParsedElement get(@NotNull String key) {
110        if (!this.object.containsKey(key)) {
111            throw new NullPointerException("Key \"" + key + "\" does not exist on this object");
112        }
113        return this.object.get(key);
114    }
115
116    /**
117     * Get the value of a certain key this object stores even if it doesn't exist
118     *
119     * @param key The key to get
120     *
121     * @return The value of a certain key this object stores or null if it doesn't exist
122     *
123     * @since 2.0.0
124     */
125    public @NotNull ParsedElement getOrNull(@NotNull String key) {
126        return this.object.get(key);
127    }
128
129    /**
130     * Set the value of a certain key this object stores
131     *
132     * @param key   The key to set
133     * @param value The value to set
134     *
135     * @since 1.0.0
136     */
137    public void set(@NotNull String key, @NotNull ParsedElement value) {
138        this.object.put(key, value);
139    }
140
141    /**
142     * Remove a certain key/value pair this object stores
143     *
144     * @param key The key to remove
145     *
146     * @since 1.0.0
147     */
148    public void remove(@NotNull String key) {
149        this.object.remove(key);
150    }
151
152    @Override
153    public @NotNull String toString() {
154        return this.getClass().getSimpleName() + "[object=" + this.object + "]";
155    }
156
157    @Override
158    public boolean equals(@Nullable Object obj) {
159        if (obj == this) {
160            return true;
161        }
162        if (obj == null || obj.getClass() != this.getClass()) {
163            return false;
164        }
165        return ((ParsedObject) obj).object.equals(this.object);
166    }
167
168    @Override
169    public int hashCode() {
170        return this.object.hashCode();
171    }
172
173    /**
174     * Create a new empty {@link ParsedObject}
175     *
176     * @return A new empty {@link ParsedObject}
177     *
178     * @since 1.0.0
179     */
180    public static @NotNull ParsedObject create() {
181        return new ParsedObject(new LinkedHashMap<>());
182    }
183
184    /**
185     * Create a new {@link ParsedObject} that will be populated with the passed map
186     *
187     * @param object The map to use to populate the new object
188     *
189     * @return A new {@link ParsedObject} populated with the passed map
190     *
191     * @since 1.0.0
192     */
193    public static @NotNull ParsedObject from(@NotNull Map<String, ParsedElement> object) {
194        return new ParsedObject(new LinkedHashMap<>(object));
195    }
196}
