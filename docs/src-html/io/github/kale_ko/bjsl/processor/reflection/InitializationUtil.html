001package io.github.kale_ko.bjsl.processor.reflection;
002
003import io.github.kale_ko.bjsl.processor.exception.InitializationException;
004import java.lang.reflect.Array;
005import java.lang.reflect.Constructor;
006import java.lang.reflect.InvocationTargetException;
007import java.lang.reflect.Modifier;
008import org.jetbrains.annotations.NotNull;
009
010/**
011 * A utility class for initializing objects and arrays from a class type
012 *
013 * @version 2.0.0
014 * @since 1.7.0
015 */
016@SuppressWarnings("unchecked")
017public final class InitializationUtil {
018    private InitializationUtil() {
019    }
020
021    private static boolean allowInitializingNonStaticMemberParents = false;
022
023    /**
024     * Allow initializing non-static member parents
025     *
026     * @since 2.0.0
027     */
028    public static void unsafeAINSMP() {
029        allowInitializingNonStaticMemberParents = true;
030    }
031
032    /**
033     * Initialize a class safely using a 0-args constructor
034     *
035     * @param clazz The class to initialize
036     * @param <T>   The type to initialize
037     *
038     * @return The initialized instance of clazz
039     */
040    public static <T> @NotNull T initialize(@NotNull Class<T> clazz) {
041        try {
042            if (clazz.isMemberClass() && !Modifier.isStatic(clazz.getModifiers())) {
043                if (!allowInitializingNonStaticMemberParents) {
044                    throw new InitializationException(new RuntimeException("Refusing to initialize a non-static member's parent, it is likely you forgot to add the 'static' keyword to your class. If this is not the case call `InitializationUtil#unsafeAINSMP()`"));
045                }
046
047                // Non-static member constructor
048                for (Constructor<?> constructor : clazz.getDeclaredConstructors()) {
049                    if ((constructor.canAccess(null) || constructor.trySetAccessible()) && constructor.getParameterTypes().length == 1) {
050                        Object object = initialize(constructor.getParameterTypes()[0]);
051                        return (T) constructor.newInstance(object);
052                    }
053                }
054            } else {
055                // Normal constructor
056                for (Constructor<?> constructor : clazz.getDeclaredConstructors()) {
057                    if ((constructor.canAccess(null) || constructor.trySetAccessible()) && constructor.getParameterTypes().length == 0) {
058                        return (T) constructor.newInstance();
059                    }
060                }
061            }
062        } catch (InstantiationException | IllegalAccessException | IllegalArgumentException | InvocationTargetException | SecurityException e) {
063            throw new InitializationException(e);
064        }
065
066        throw new InitializationException(clazz);
067    }
068
069    /**
070     * Initialize an array safely using {@link Array#newInstance(Class, int)}
071     *
072     * @param clazz  The class to initialize
073     * @param length The length of the new array
074     * @param <T>    The type to initialize
075     *
076     * @return The initialized array instance of clazz with the passed length
077     */
078    public static <T> @NotNull T[] initializeArray(@NotNull Class<T> clazz, int length) {
079        return (T[]) Array.newInstance(clazz, length);
080    }
081
082    /**
083     * Initialize a primitive array safely using {@link Array#newInstance(Class, int)}
084     *
085     * @param clazz  The class to initialize
086     * @param length The length of the new array
087     * @param <T>    The type to initialize
088     *
089     * @return The initialized array instance of clazz with the passed length
090     */
091    public static <T> @NotNull Object initializePrimitiveArray(@NotNull Class<T> clazz, int length) {
092        return Array.newInstance(clazz, length);
093    }
094}
