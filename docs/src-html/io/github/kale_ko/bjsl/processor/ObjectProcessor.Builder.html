702    /**
703     * Maps this element into an Object
704     *
705     * @param element The element to map
706     * @param type    The object type to map to
707     *
708     * @return A new Object of passed type with the values of element
709     *
710     * @throws io.github.kale_ko.bjsl.processor.exception.ProcessorException If there is an exception while processing
711     * @since 1.0.0
712     */
713    @SuppressWarnings("unchecked")
714    public @Nullable Object toObject(@NotNull ParsedElement element, @NotNull JavaType type) {
715        try {
716            if (element.isPrimitive() && element.asPrimitive().isNull()) {
717                return null;
718            }
719
720            {
721                Optional<TypeProcessor> typeProcessor = typeProcessors.entrySet().stream().filter(entry -> TypeUtils.isTypeOrSuperTypeOf(type, entry.getKey())).sorted((entryA, entryB) -> TypeUtils.sortDistance(type, entryA.getKey(), entryB.getKey())).map(Map.Entry::getValue).findFirst();
722                if (typeProcessor.isPresent()) {
723                    return typeProcessor.get().toObject(element);
724                }
725            }
726
727            if (element instanceof ParsedPrimitive) {
728                if (type.getRawClass().isEnum()) {
729                    if (element.asPrimitive().isString()) {
730                        for (Object value : type.getRawClass().getEnumConstants()) {
731                            if ((!caseSensitiveEnums && ((Enum<?>) value).name().equalsIgnoreCase(element.asPrimitive().asString())) || (caseSensitiveEnums && ((Enum<?>) value).name().equals(element.asPrimitive().asString()))) {
732                                return value;
733                            }
734                        }
735
736                        if (BJSL.getLogger() != null) {
737                            BJSL.getLogger().warning("Unknown enum value \"" + element.asPrimitive().asString() + "\" for type \"" + type.getRawClass().getSimpleName() + "\"");
738                        }
739
740                        return null;
741                    } else if (element.asPrimitive().isNull()) {
742                        return null;
743                    } else {
744                        throw new EnumExpectedException(type.getRawClass());
745                    }
746                } else {
747                    Object object = element.asPrimitive().get();
748
749                    if (object == null) {
750                        return null;
751                    } else if (type.getRawClass() == String.class) {
752                        return object.toString();
753                    } else if (type.getRawClass() == Byte.class || type.getRawClass() == byte.class) {
754                        if (object.getClass() == String.class) {
755                            return Byte.parseByte((String) object);

1286                    if (!((ignoreNulls && (subElement.isPrimitive() && subElement.asPrimitive().isNull())) || (ignoreEmptyObjects && (subElement.isObject() && subElement.asObject().getSize() == 0) || (subElement.isArray() && subElement.asArray().getSize() == 0)))) {

1297                    if (!((ignoreNulls && (subElement.isPrimitive() && subElement.asPrimitive().isNull())) || (ignoreEmptyObjects && (subElement.isObject() && subElement.asObject().getSize() == 0) || (subElement.isArray() && subElement.asArray().getSize() == 0)))) {

1307                    ParsedElement subElement = toElement(entry.getValue());
1308                    if (!((ignoreNulls && (subElement.isPrimitive() && subElement.asPrimitive().isNull())) || (ignoreEmptyObjects && (subElement.isObject() && subElement.asObject().getSize() == 0) || (subElement.isArray() && subElement.asArray().getSize() == 0)))) {

1330                    }
1331                }
1332

1334                for (Field field : fields) {
1335                    if (!Modifier.isStatic(field.getModifiers()) && (field.canAccess(object) || field.trySetAccessible())) {

1337
1338                        String subKey = field.getName();
1339                        ParsedElement subElement = toElement(field.get(object));
1340
1341                        if (ignoreNulls && (subElement.isPrimitive() && subElement.asPrimitive().isNull())) {
1342                            shouldSerialize = false;
1343                        }
1344
1345                        if (ignoreEmptyObjects && ((subElement.isObject() && subElement.asObject().getSize() == 0) || (subElement.isArray() && subElement.asArray().getSize() == 0))) {
1346                            shouldSerialize = false;
1347                        }
1348
1349                        if (ignoreDefaults && (subElement.isPrimitive() && (!subElement.asPrimitive().isNull() ? subElement.asPrimitive().get().equals(field.get(defaultObject)) : field.get(defaultObject) == null))) {
1350                            shouldSerialize = false;
1351                        }
1352
1353                        for (Annotation annotation : field.getDeclaredAnnotations()) {
1354                            if (annotation.annotationType() == AlwaysSerialize.class) {
1355                                shouldSerialize = true;
1356                            } else if (annotation.annotationType() == NeverSerialize.class) {
1357                                shouldSerialize = false;
1358                            } else if (annotation.annotationType() == Rename.class) {
1359                                subKey = ((Rename) annotation).value();
1360                            }
1361                        }
1362
1363                        if (shouldSerialize) {
1364                            objectElement.set(subKey, subElement);
1365                        }
1366                    }
1367                }
1368
1369                return objectElement;
1370            }
1371        } catch (Exception e) {
1372            throw new ProcessorException(e);
1373        }

1400                    case CHAR: {

1403                    case SHORT: {

1406                    case INTEGER: {

1409                    case LONG: {

1411                    }
1412                    case BIGINTEGER: {

1414                    }
1415                    case FLOAT: {
1416                        return Float.toString(primitive.asFloat());
1417                    }
1418                    case DOUBLE: {

1420                    }
1421                    case BIGDECIMAL: {
1422                        return primitive.asBigDecimal().toString();
1423                    }
1424                    case BOOLEAN: {
1425                        return Boolean.toString(primitive.asBoolean());
1426                    }
1427                    case NULL: {
1428                        return "null";
1429                    }
1430                    default: {
1431                        throw new NotAPrimitiveException(element);

1434            } else {
1435                throw new NotAPrimitiveException(element);

1445    /**
1446     * Get all the fields on a class and its superclasses
1447     *
1448     * @param clazz The class to get the fields of
1449     * @param <T>   The class to get the fields of
1450     *
1451     * @return All the fields on the class and its superclasses
1452     *
1453     * @since 1.0.0
1454     */
1455    protected static <T> @NotNull List<Field> getFields(@NotNull Class<T> clazz) {
1456        Set<String> fieldNames = new HashSet<>();
1457        List<Field> fields = new ArrayList<>(Arrays.asList(clazz.getDeclaredFields()));
1458
1459        Class<?> superClazz = clazz.getSuperclass();
1460        if (superClazz != null && superClazz != Object.class) {
1461            List<Field> superFields = getFields(superClazz);
1462
1463            for (Field superField : superFields) {
1464                boolean overwritten = fieldNames.contains(superField.getName());
1465
1466                if (!(overwritten || Modifier.isTransient(superField.getModifiers()))) {
1467                    fieldNames.add(superField.getName());
1468                    fields.add(superField);
1469                }
1470            }
1471        }
1472
1473        return fields;
1474    }
