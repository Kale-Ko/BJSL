281    /**
282     * A type processor for {@link java.util.UUID}. Converts to and from a ParsedPrimitive string, array, or bigint
283     *
284     * @since 2.0.0
285     */
286    public final @NotNull TypeProcessor UUID_P = new TypeProcessor() {
287        @Override
288        public @NotNull ParsedElement toElement(@Nullable Object object) {
289            if (object == null) {
290                return ParsedPrimitive.fromNull();
291            }
292
293            if (object instanceof UUID) {
294                UUID uuid = (UUID) object;
295
296                switch (options.getUuidMode()) {
297                    case STRING: {
298                        return ParsedPrimitive.fromString(uuid.toString());
299                    }
300                    case BYTE_ARRAY: {
301                        ByteBuffer byteBuffer = ByteBuffer.wrap(new byte[16]);
302                        byteBuffer.putLong(uuid.getMostSignificantBits());
303                        byteBuffer.putLong(uuid.getLeastSignificantBits());
304
305                        ParsedArray byteArray = ParsedArray.create();
306                        byte[] array = new byte[16];
307                        byteBuffer.position(0).get(array);
308                        for (byte b : array) {
309                            byteArray.add(ParsedPrimitive.fromByte(b));
310                        }
311                        return byteArray;
312                    }
313                    case SHORT_ARRAY: {
314                        ByteBuffer shortBuffer = ByteBuffer.wrap(new byte[16]);
315                        shortBuffer.putLong(uuid.getMostSignificantBits());
316                        shortBuffer.putLong(uuid.getLeastSignificantBits());
317
318                        ParsedArray shortArray = ParsedArray.create();
319                        short[] array = new short[8];
320                        shortBuffer.position(0).asShortBuffer().get(array);
321                        for (short s : array) {
322                            shortArray.add(ParsedPrimitive.fromShort(s));
323                        }
324                        return shortArray;
325                    }
326                    case INT_ARRAY: {
327                        ByteBuffer intBuffer = ByteBuffer.wrap(new byte[16]);
328                        intBuffer.putLong(uuid.getMostSignificantBits());
329                        intBuffer.putLong(uuid.getLeastSignificantBits());
330
331                        ParsedArray intArray = ParsedArray.create();
332                        int[] array = new int[4];
333                        intBuffer.position(0).asIntBuffer().get(array);
334                        for (int i : array) {
335                            intArray.add(ParsedPrimitive.fromInteger(i));
336                        }
337                        return intArray;
338                    }
339                    case LONG_ARRAY: {
340                        ByteBuffer longBuffer = ByteBuffer.wrap(new byte[16]);
341                        longBuffer.putLong(uuid.getMostSignificantBits());
342                        longBuffer.putLong(uuid.getLeastSignificantBits());
343
344                        ParsedArray longArray = ParsedArray.create();
345                        long[] array = new long[2];
346                        longBuffer.position(0).asLongBuffer().get(array);
347                        for (long l : array) {
348                            longArray.add(ParsedPrimitive.fromLong(l));
349                        }
350                        return longArray;
351                    }
352                    case NUMBER: {
353                        ByteBuffer bigIntBuffer = ByteBuffer.wrap(new byte[16]);
354                        bigIntBuffer.putLong(uuid.getMostSignificantBits());
355                        bigIntBuffer.putLong(uuid.getLeastSignificantBits());
356
357                        return ParsedPrimitive.fromBigInteger(new BigInteger(bigIntBuffer.array()));
358                    }
359                    default: {
360                        throw new RuntimeException();
361                    }
362                }
363            } else {
364                throw new InvalidParameterException("object must be UUID");
365            }
366        }
367
368        @Override
369        public @Nullable Object toObject(@NotNull ParsedElement element) {
370            if (element.isPrimitive() && element.asPrimitive().isNull()) {
371                return null;
372            }
373
374            if (element.isArray()) {
375                ByteBuffer buffer = ByteBuffer.wrap(new byte[16]);
376
377                for (ParsedElement subElement : element.asArray().getValues()) {
378                    if (subElement.isPrimitive()) {
379                        switch (element.asArray().getSize()) {
380                            case 16: {
381                                switch (subElement.asPrimitive().getType()) {
382                                    case BYTE: {
383                                        buffer.put(subElement.asPrimitive().asByte());
384                                        break;
385                                    }
386                                    case SHORT: {
387                                        buffer.put((byte) subElement.asPrimitive().asShort());
388                                        break;
389                                    }
390                                    case INTEGER: {
391                                        buffer.put((byte) subElement.asPrimitive().asInteger());
392                                        break;
393                                    }
394                                    case LONG: {
395                                        buffer.put((byte) subElement.asPrimitive().asLong());
396                                        break;
397                                    }
398                                    default: {
399                                        throw new RuntimeException();
400                                    }
401                                }
402                                break;
403                            }
404                            case 8: {
405                                switch (subElement.asPrimitive().getType()) {
406                                    case BYTE: {
407                                        buffer.putShort(subElement.asPrimitive().asByte());
408                                        break;
409                                    }
410                                    case SHORT: {
411                                        buffer.putShort(subElement.asPrimitive().asShort());
412                                        break;
413                                    }
414                                    case INTEGER: {
415                                        buffer.putShort((short) subElement.asPrimitive().asInteger());
416                                        break;
417                                    }
418                                    case LONG: {
419                                        buffer.putShort((short) subElement.asPrimitive().asLong());
420                                        break;
421                                    }
422                                    default: {
423                                        throw new RuntimeException();
424                                    }
425                                }
426                                break;
427                            }
428                            case 4: {
429                                switch (subElement.asPrimitive().getType()) {
430                                    case BYTE: {
431                                        buffer.putInt(subElement.asPrimitive().asByte());
432                                        break;
433                                    }
434                                    case SHORT: {
435                                        buffer.putInt(subElement.asPrimitive().asShort());
436                                        break;
437                                    }
438                                    case INTEGER: {
439                                        buffer.putInt(subElement.asPrimitive().asInteger());
440                                        break;
441                                    }
442                                    case LONG: {
443                                        buffer.putInt((int) subElement.asPrimitive().asLong());
444                                        break;
445                                    }
446                                    default: {
447                                        throw new RuntimeException();
448                                    }
449                                }
450                                break;
451                            }
452                            case 2: {
453                                switch (subElement.asPrimitive().getType()) {
454                                    case BYTE: {
455                                        buffer.putLong(subElement.asPrimitive().asByte());
456                                        break;
457                                    }
458                                    case SHORT: {
459                                        buffer.putLong(subElement.asPrimitive().asShort());
460                                        break;
461                                    }
462                                    case INTEGER: {
463                                        buffer.putLong(subElement.asPrimitive().asInteger());
464                                        break;
465                                    }
466                                    case LONG: {
467                                        buffer.putLong(subElement.asPrimitive().asLong());
468                                        break;
469                                    }
470                                    default: {
471                                        throw new RuntimeException();
472                                    }
473                                }
474                                break;
475                            }
476                            default: {
477                                throw new RuntimeException();
478                            }
479                        }
480                    } else {
481                        throw new RuntimeException();
482                    }
483                }
484
485                buffer = buffer.position(0);
486
487                return new UUID(buffer.getLong(), buffer.getLong());
488            } else if (element.isPrimitive()) {
489                switch (element.asPrimitive().getType()) {
490                    case STRING: {
491                        return UUID.fromString(element.asPrimitive().asString());
492                    }
493                    case BYTE: {
494                        return new UUID(0, (long) element.asPrimitive().asByte());
495                    }
496                    case SHORT: {
497                        return new UUID(0, (long) element.asPrimitive().asShort());
498                    }
499                    case INTEGER: {
500                        return new UUID(0, (long) element.asPrimitive().asInteger());
501                    }
502                    case LONG: {
503                        return new UUID(0, element.asPrimitive().asLong());
504                    }
505                    case BIGINTEGER: {
506                        BigInteger bigInt = element.asPrimitive().asBigInteger();
507                        return new UUID(bigInt.shiftRight(64).longValue(), bigInt.longValue());
508                    }
509                    default: {
510                        throw new InvalidParameterException("object must be Array, BigInt, or String");
511                    }
512                }
513            } else {
514                throw new InvalidParameterException("object must be Array, BigInt, or String");
515            }
516        }
517    };

660     * A type processor for {@link java.net.InetAddress}. Converts to and from a ParsedPrimitive string or bigint
661     *
662     * @since 2.0.0
663     */
664    public final @NotNull TypeProcessor INET_ADDRESS_P = new TypeProcessor() {
665        final BigInteger MASK_8 = BigInteger.TWO.pow(8).subtract(BigInteger.ONE);
666        final BigInteger MASK_32 = BigInteger.TWO.pow(32).subtract(BigInteger.ONE);
667        final BigInteger MASK_128 = BigInteger.TWO.pow(128).subtract(BigInteger.ONE);
668
669        @Override
670        public @NotNull ParsedElement toElement(@Nullable Object object) {
671            if (object == null) {
672                return ParsedPrimitive.fromNull();
673            }
674
675            if (object instanceof InetAddress) {
676                boolean hasHostname = !((InetAddress) object).toString().startsWith("/"); // Kinda hacky but the only way
677
678                // This works but is illegal in newer java versions
679                // try {
680                //     Field holderField = InetAddress.class.getDeclaredField("holder");
681                //     holderField.setAccessible(true);
682                //     Object holder = holderField.get(((InetAddress) object));
683                //
684                //     Field hostnameField = holder.getClass().getDeclaredField("hostName");
685                //     hostnameField.setAccessible(true);
686                //
687                //     hasHostname = hostnameField.get(holder) != null;
688                // } catch (IllegalAccessException | NoSuchFieldException e) {
689                //     throw new RuntimeException(e);
690                // }
691
692                if (!hasHostname) {
693                    byte[] address = ((InetAddress) object).getAddress();
694
695                    switch (options.getInetAddressMode()) {
696                        case STRING: {
697                            if (address.length == 4) {
698                                String[] addressBytes = new String[4];
699
700                                for (int i = 0; i < 4; i++) {
701                                    addressBytes[i] = Integer.toUnsignedString(address[i] & 0xFF, 10);
702                                    while (options.isFillAddresses() && addressBytes[i].length() < 3) {
703                                        addressBytes[i] = "0" + addressBytes[i];
704                                    }
705                                }
706
707                                return ParsedPrimitive.fromString(String.join(".", addressBytes));
708                            } else if (address.length == 16) {
709                                StringBuilder stringBuilder = new StringBuilder();
710
711                                if (Arrays.equals(address, new byte[address.length])) {
712                                    return ParsedPrimitive.fromString("::");
713                                }
714
715                                int currentIndex = -1, currentLength = 0;
716                                int maxIndex = -1, maxLength = 0;
717
718                                for (int i = 0; i < 8; i++) {
719                                    if ((((address[i * 2] & 0xFF) << 8) + (address[(i * 2) + 1] & 0xFF)) == 0) {
720                                        if (currentLength == 0) {
721                                            currentIndex = i;
722                                        }
723
724                                        currentLength++;
725                                    } else {
726                                        if (currentLength > maxLength) {
727                                            maxIndex = currentIndex;
728                                            maxLength = currentLength;
729                                        }
730
731                                        currentLength = 0;
732                                    }
733                                }
734
735                                if (currentLength > maxLength) {
736                                    maxIndex = currentIndex;
737                                    maxLength = currentLength;
738                                }
739
740                                boolean first = true;
741
742                                for (int i = 0; i < 8; i++) {
743                                    if (!options.isFillAddresses() && i == maxIndex) {
744                                        stringBuilder.append(":");
745                                        if (first) {
746                                            stringBuilder.append(":");
747                                            first = false;
748                                        }
749                                        i += maxLength - 1;
750                                    } else {
751                                        StringBuilder subStringBuilder = new StringBuilder(Integer.toUnsignedString(((address[i * 2] & 0xFF) << 8) + (address[(i * 2) + 1] & 0xFF), 16));
752                                        while (options.isFillAddresses() && subStringBuilder.length() < 4) {
753                                            subStringBuilder.insert(0, "0");
754                                        }
755                                        stringBuilder.append(subStringBuilder);
756
757                                        if (i < 7) {
758                                            stringBuilder.append(":");
759                                        }
760
761                                        first = false;
762                                    }
763                                }
764
765                                return ParsedPrimitive.fromString(stringBuilder.toString());
766                            } else {
767                                throw new InvalidParameterException("InetAddress must be IPv4 or IPv6");
768                            }
769                        }
770                        case NUMBER: {
771                            byte[] extendedAddress = new byte[address.length + 1]; // Avoid our number being interpreted as a signed number
772                            System.arraycopy(address, 0, extendedAddress, 1, address.length);
773
774                            if (address.length == 4) {
775                                return ParsedPrimitive.fromBigInteger(new BigInteger(extendedAddress));
776                            } else if (address.length == 16) {
777                                return ParsedPrimitive.fromBigInteger(new BigInteger(extendedAddress).or(BigInteger.ONE.shiftLeft(128)));
778                            } else {
779                                throw new InvalidParameterException("InetAddress must be IPv4 or IPv6");
780                            }
781                        }
782                        default: {
783                            throw new RuntimeException();
784                        }
785                    }
786                } else {
787                    switch (options.getInetAddressMode()) {
788                        case STRING: {
789                            return ParsedPrimitive.fromString(((InetAddress) object).getHostName());
790                        }
791                        case NUMBER: {
792                            throw new InvalidParameterException("Cannot convert hostname to Number");
793                        }
794                        default: {
795                            throw new RuntimeException();
796                        }
797                    }
798                }
799            } else {
800                throw new InvalidParameterException("object must be InetAddress");
801            }
802        }

804        @Override
805        public @Nullable Object toObject(@NotNull ParsedElement element) {
806            if (element.isPrimitive() && element.asPrimitive().isNull()) {
807                return null;
808            }
809
810            if (element.isPrimitive()) {
811                switch (element.asPrimitive().getType()) {
812                    case STRING: {
813                        try {
814                            String addressString = element.asPrimitive().asString();
815                            if (addressString.startsWith("[") && addressString.endsWith("]")) {
816                                addressString = addressString.substring(1, addressString.length() - 1);
817                            }
818
819                            if (addressString.equalsIgnoreCase("::")) {
820                                return InetAddress.getByAddress(new byte[16]);
821                            }
822                            if (addressString.startsWith("::")) {
823                                addressString = addressString.substring(1);
824                            }
825                            if (addressString.endsWith("::")) {
826                                addressString = addressString.substring(0, addressString.length() - 1);
827                            }
828
829                            String[] addressBytes = addressString.split("[.:]", -1);
830
831                            if (addressBytes.length == 4 && addressString.contains(".") && !addressString.contains(":")) {
832                                byte[] address = new byte[4];
833
834                                for (int i = 0; i < 4; i++) {
835                                    address[i] = (byte) (Integer.parseUnsignedInt(addressBytes[i], 10) & 0xFF);
836                                }
837
838                                return InetAddress.getByAddress(address);
839                            } else if (addressString.contains(":") && !addressString.contains(".")) {
840                                byte[] address = new byte[16];
841
842                                for (int i = 0, j = 0; i < 8; i++) {
843                                    if (addressBytes[i + j].isEmpty()) {
844                                        int length = 8 - addressBytes.length;
845                                        i += length;
846                                        j -= length;
847                                        continue;
848                                    }
849
850                                    address[(i) * 2] = (byte) ((Integer.parseUnsignedInt(addressBytes[i + j], 16) >> 8) & 0xFF);
851                                    address[((i) * 2) + 1] = (byte) (Integer.parseUnsignedInt(addressBytes[i + j], 16) & 0xFF);
852                                }
853
854                                return InetAddress.getByAddress(address);
855                            } else {
856                                throw new InvalidParameterException("InetAddress is not IPv4 or IPv6");
857                            }
858                        } catch (UnknownHostException e) {
859                            throw new RuntimeException("object is an invalid InetAddress", e);
860                        } catch (InvalidParameterException e) {
861                            try {
862                                return InetAddress.getByName(element.asPrimitive().asString());
863                            } catch (UnknownHostException e2) {
864                                throw new RuntimeException("object is an invalid InetAddress", e2);
865                            }
866                        }
867                    }
868                    case BYTE:
869                        ByteBuffer byteBuffer = ByteBuffer.wrap(new byte[4]);
870                        byteBuffer.put(element.asPrimitive().asByte());
871
872                        try {
873                            return InetAddress.getByAddress(byteBuffer.array());
874                        } catch (UnknownHostException e) {
875                            throw new InvalidParameterException("object is an invalid InetAddress");
876                        }
877                    case SHORT:
878                        ByteBuffer shortBuffer = ByteBuffer.wrap(new byte[4]);
879                        shortBuffer.putShort(element.asPrimitive().asShort());
880
881                        try {
882                            return InetAddress.getByAddress(shortBuffer.array());
883                        } catch (UnknownHostException e) {
884                            throw new InvalidParameterException("object is an invalid InetAddress");
885                        }
886                    case INTEGER:
887                        ByteBuffer intBuffer = ByteBuffer.wrap(new byte[4]);
888                        intBuffer.putInt(element.asPrimitive().asInteger());
889
890                        try {
891                            return InetAddress.getByAddress(intBuffer.array());
892                        } catch (UnknownHostException e) {
893                            throw new InvalidParameterException("object is an invalid InetAddress");
894                        }
895                    case LONG:
896                        ByteBuffer longBuffer = ByteBuffer.wrap(new byte[4]);
897                        longBuffer.putLong(element.asPrimitive().asLong());
898
899                        try {
900                            return InetAddress.getByAddress(longBuffer.array());
901                        } catch (UnknownHostException e) {
902                            throw new InvalidParameterException("object is an invalid InetAddress");
903                        }
904                    case BIGINTEGER: {
905                        BigInteger bigInt = element.asPrimitive().asBigInteger();
906                        BigInteger upper = bigInt.shiftRight(128);
907
908                        boolean isIPv4 = upper.and(MASK_8).equals(BigInteger.ZERO);
909
910                        BigInteger lower = bigInt.and(isIPv4 ? MASK_32 : MASK_128);
911                        byte[] address = lower.toByteArray();
912
913                        if ((isIPv4 && address.length < 4) || (!isIPv4 && address.length < 16)) {
914                            byte[] fullAddress = new byte[isIPv4 ? 4 : 16];
915                            System.arraycopy(address, 0, fullAddress, fullAddress.length - address.length, address.length);
916                            address = fullAddress;
917                        }
918                        if ((isIPv4 && address.length > 4) || (!isIPv4 && address.length > 16)) {
919                            address = Arrays.copyOfRange(address, address.length - (isIPv4 ? 4 : 16), address.length);
920                        }
921
922                        try {
923                            return InetAddress.getByAddress(address);
924                        } catch (UnknownHostException e) {
925                            throw new RuntimeException("object is an invalid InetAddress", e);
926                        }
927                    }
928                    default: {
929                        throw new InvalidParameterException("object must be String");
930                    }
931                }
932            } else {
933                throw new InvalidParameterException("object must be String");
934            }
935        }
