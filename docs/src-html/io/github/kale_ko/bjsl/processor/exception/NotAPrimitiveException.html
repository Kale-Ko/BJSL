001package io.github.kale_ko.bjsl.processor.exception;
002
003import io.github.kale_ko.bjsl.elements.ParsedElement;
004import org.jetbrains.annotations.NotNull;
005
006/**
007 * Thrown when a value is attempted to be used as a map key but is not a string or other string-able primitive
008 *
009 * @version 2.0.0
010 * @since 2.0.0
011 */
012public class NotAPrimitiveException extends RuntimeException {
013    /**
014     * Create a new NotAPrimitiveException
015     *
016     * @param parsedObject The parsed object that was invalid
017     */
018    public NotAPrimitiveException(@NotNull ParsedElement parsedObject) {
019        super("ParedElement is not a string or other string-able primitive, must be to be a map key!\n" + parsedObject);
020    }
021}
